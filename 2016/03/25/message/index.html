##简易留言板的实现
今天就是根据上一篇文章做个实例根据DOM节点

html部分

<div class="box">
    <input type="text" name="" placeholder="留言">
    <input type="button" name="" value="提交">
    <ul class="list"></ul>
</div>

css模块

*{
        margin: 0;
        padding: 0;
    }
    .box{
        width: 600px;
        min-height: 300px;
        border: 1px solid red;
        margin: 50px auto 0;
    }
    [type="text"]{
        border: 10px solid #000;
        border-radius: 20px;
        height: 40px;
        font-size: 24px;
        width: 400px;
    }
    [type="button"]{
        border: 10px solid #000;
        border-radius: 20px;
        height: 60px;
        font-size: 24px;
        width: 120px;
    }
    ul li{
        height: 40px;
        line-height: 40px;
        padding-left: 10px;
        border-bottom: 1px solid red;
        font-size: 24px;
        position: relative;
    }
    ul li:nth-child(odd){
        /*odd 奇数  evet 偶数*/
        background: lightblue;
    }
    ul li:nth-child(even){
        /*odd 奇数  evet 偶数*/
        background: cyan;
    }
     ul li div{
         width: 80px;
         height: 40px;
         background: gray;
        font-size: 24px;
        color: black;
        position: absolute;
        right: 0px;
        top: 0px;
        text-align: center;
        line-height: 40px; 
    }

js模块

<script type="text/javascript">
var Ul=document.querySelector(".list");
var btn=Ul.previousElementSibling;
btn.onclick=function(){
    var text=this.previousElementSibling.value;

    var a=document.createElement("a");
    var remove=document.createElement("div");
    a.href="#";
    var li=document.createElement("li");
    var date=new Date();
    a.innerText=date.getHours()+":"+date.getMinutes()+":"+date.getSeconds();
    li.innerText=text;
    remove.innerText="删除";
    li.appendChild(a);
    li.appendChild(remove);
    Ul.appendChild(li);
    Ul.insertBefore(li,Ul.firstChild);
    remove.onclick=function(){
        Ul.removeChild(li);
    }
}

</script>

这些代码中用到的DOM节点并不多但是是最常用的知识点很实用，这里面的逻辑也比较直接，易懂
