JS模块处理

<script>
定义变量
    var div1=document.querySelector(".div1");
    var disX=0;
    var disY=0;
    var timer=null;


通过移动的最后两个点来找速度和方向
    var prevX=0;
    var prevY=0;
    var iSpeedX=0;
    var iSpeedY=0;
鼠标按下时    
    div1.onmousedown=function(ev){
        var ev=ev||window.event;
        disX=ev.clientX-div1.offsetLeft;
        console.log(ev.clientX,div1.offsetLeft)
        disY=ev.clientY-div1.offsetTop;

        prevX=ev.clientX;
        prevY=ev.clientY
        clearInterval(timer)
        document.onmousemove=function(ev){
        速度就是两个点得差值
            var ev=ev||window.event;
            div1.style.left=ev.clientX-disX+"px";
            div1.style.top=ev.clientY-disY+"px";
            iSpeedX=ev.clientX-prevX;
            iSpeedY=ev.clientY-prevY;
        一直保持最后两个点得差值
            prevX=ev.clientX;
            prevY=ev.clientY;
        }
        document.onmouseup=function(){
            document.onmousemove=null;
            document.onmouseup=null;
            startMove();
        }
    让小球移动的函数
    function startMove(){
        clearInterval(timer)
        timer=setInterval(function(){
        // 自由落体
        iSpeedY+=3;
        var L=div1.offsetLeft+iSpeedX;
        var T=div1.offsetTop+iSpeedY;
        if(L<0){
            L=0;
            // 控制方向
            iSpeedX*=-1;
            // 摩擦系数
            iSpeedX*=0.75;
        }else if(L>document.documentElement.clientWidth-div1.offsetWidth){
            L=document.documentElement.clientWidth-div1.offsetWidth;
            iSpeedX*=-1;
            iSpeedY*=0.75;
            iSpeedX*=0.75;
        }
        if(T<0){
            T=0;
            iSpeedY*=-1;
        }else if(T>document.documentElement.clientHeight-div1.offsetHeight){
            T=document.documentElement.clientHeight-div1.offsetHeight;
            iSpeedY*=-1
        }
        div1.style.left=L+"px";
        div1.style.top=T+"px";
    },30)
}
</script>

上述代码的实现是对于一些简单的js逻辑上的处理 比如对clientX,clientY,以及对dom的一些使用，我个人认为这里面最值得学习的是，特别是针对新手，很重要，这里面的碰撞检测是大多数游戏中或是一些JS实例中最容易体现，希望采纳
