数据库添加函数

function add($sql){
    $link=mysqli_connect("localhost","root","root","db");
    mysqli_query($link,"set names utf8");
    $res=mysqli_query($link,$sql);
    if($res){
        return true;
    }else{
        return false;
    }
}
$sql="INSERT INTO users (username,age,pwd) VALUES ('王健林',40,123000)";
 $result=add($sql);
if($result){
echo "添加成功";
 }else{
    echo "添加失败";
}
